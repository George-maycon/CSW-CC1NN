import random

    def jogar_adivinhacao():
        print("Bem-vindo ao Jogo de Adivinhação!")
    
        # Número aleatório entre 1 e 100
        numero_secreto = random.randint(1, 100)
        tentativas = 0
        acertou = False
    
        # Explicação para o usuário
        print("Estou pensando em um número entre 1 e 100. Tente adivinhar!")
    
        # Loop para o jogador tentar adivinhar
        while not acertou:
            tentativas += 1
            palpite = int(input(f"Tentativa {tentativas}: Qual é o seu palpite? "))
    
            # Verificando o palpite
            if palpite < numero_secreto:
                print("O número é maior. Tente novamente.")
            elif palpite > numero_secreto:
                print("O número é menor. Tente novamente.")
            else:
                acertou = True
                print(f"Parabéns! Você acertou o número {numero_secreto} em {tentativas} tentativas!")
    
    # Iniciar o jogo
    if __name__ == "__main__":
        jogar_adivinhacao()
